function simpanNilai(e){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Nilai"); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.nis, data.nama, data.tahfidz, data.murojaah, data.catatan ]); return ContentService .createTextOutput( JSON.stringify({ status:"nilai tersimpan" }) ) .setMimeType(ContentService.MimeType.JSON); }
