// ================================= // TAHFIDZKU API // ================================= // =============================== // AMBIL DATA // =============================== function doGet(e){ var tipe = e.parameter.tipe; // =================== // SANTRI // =================== if(tipe=="santri"){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Santri"); return outputJSON( sheet.getDataRange().getValues() ); } // =================== // NILAI // =================== if(tipe=="nilai"){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Nilai"); return outputJSON( sheet.getDataRange().getValues() ); } // =================== // MUROJAAH // =================== if(tipe=="murojaah"){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Murojaah"); return outputJSON( sheet.getDataRange().getValues() ); } return outputJSON({ status:"TahfidzKu API aktif" }); } // =============================== // SIMPAN DATA // =============================== function doPost(e){ var data = JSON.parse(e.postData.contents); // =================== // SANTRI // =================== if(data.tipe=="santri"){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Santri"); sheet.appendRow([ data.nis, data.nama, data.kelas, data.halaqah, data.target, data.jumlah ]); return outputJSON({ status:"Santri tersimpan" }); } // =================== // NILAI // =================== if(data.tipe=="nilai"){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Nilai"); sheet.appendRow([ data.nis, data.nama, data.tahfidz, data.murojaah, data.catatan ]); return outputJSON({ status:"Nilai tersimpan" }); } // =================== // MUROJAAH // =================== if(data.tipe=="murojaah"){ var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Murojaah"); sheet.appendRow([ data.tanggal, data.nis, data.nama, data.juz, data.status, data.catatan ]); return outputJSON({ status:"Murojaah tersimpan" }); } return outputJSON({ status:"error", pesan:"Tipe tidak ditemukan" }); } // =============================== // JSON OUTPUT // =============================== function outputJSON(data){ return ContentService .createTextOutput( JSON.stringify(data) ) .setMimeType( ContentService.MimeType.JSON ); }
